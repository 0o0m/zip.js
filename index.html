zip.js

A new JavasScript library to zip and unzip files

zip.js offers a low-level API for writing and reading (big) zip files

This library is implemented with these HTML5 features:

  • Web workers
  • FileReaderSync and FileWriter
  • Blob and BlobBuilder
  • Typed arrays

Notes

  • This library is currently only supported with chrome but will work with firefox when needed API will be implemented.
  • A Chrome issue prevents the library to manage efficiently big files into a zip

Install

  • include zip.js, crc32.js, deflate.js, and inflate.js file in your project
  • add zip.js in your HTML page 
    <script type="text/javascript" src="/lib/zip.js"></script>
  • set WORKER_SCRIPTS_PATH value in zip.js if needed 
    var WORKER_SCRIPTS_PATH = "/lib/";

API

  1. Write a zip into a File object

    • get a FileWriter object to write the zip into a File object 
      // fileWriter is used in add method example
var fileWriter;
zip.createFileWriter(outputFile, function(writer) {
  fileWriter = writer;
  // ...

}, function(error) {
  // onerror callback
});
    • add a file as a Blob object into the zip 
      // Use the add method of the FileWriter object to add the blob
fileWriter.add(filename, blob, null /* options */, function() {

  // onend callback

}, function(currentIndex, totalIndex) {
  // onprogress callback
}, function(error) {
  // onerror callback
});
    • close the zip 
      fileWriter.close(function() {

  // onclose callback

}, function(error) {
  // onerror callback
});

  2. Read a zip from a Blob object

    • get a BlobReader object to read a zip from a Blob object 
      // zipReader is used in getEntries method example
var zipReader;
zip.createBlobReader(blob, function(reader) {
  zipReader = reader;
  // ...

}, function(error) {
  // onerror callback
});
    • get a HttpRangeReader object to read a zip file from a given URL 
      // zipReader is used in getEntries method example
var zipReader;
httpResourceReader.init(URL, function(reader) {
  zipReader = reader;
  // ...

}, function(error) {
  // onerror callback
});
    • get all entries from the zip 
      // Use the getEntries method of the BlobReader or HttpRangeReader object
// entry is used is used in getData method example
var entry;
zipReader.getEntries(function(entries) {

  // entries parameter is an Array of Entry objects

}, function(error) {
  // onerror callback
});
    • get entry content as a Blob object 
      // Use the getData method of an Entry object
entry.getData(function(blob) {

  // data parameter contains the file entry content

}, function(current, total) {
  // onprogress callback
}, function(error) {
  // onerror callback
});
    • close the zip 
      zipReader.close(function() {

  // onclose callback

}, function(error) {
  // onerror callback
});
