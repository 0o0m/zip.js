zip.js

A JavasScript library to zip and unzip files

zip.js offers a low-level API for writing and reading large zip files (up to 4GB) with a stable RAM use.

This library is implemented with these HTML5 features:

  • Web workers
  • FileReader (and FileWriter optionally)
  • Blob and BlobBuilder
  • Typed arrays

Notes

  • This library works currently only with Chrome and Firefox (with DataView.js from David Flanagan).

Install

  • include zip.js, deflate.js, inflate.js and dataview.js (for Firefox DataView support) in your project
  • add zip.js in your HTML page 
    <script type="text/javascript" src="/lib/zip.js"></script>
  • set zip.workersScriptPath if needed 
    zip.workersScriptPath = "/lib/";

Examples

  1. Write a zip into a File object

    • get a ZipWriter object to write into the zip 
      var zipWriter; // a reference to the ZipWriter object
// use a FileWriter to store the zip into a FileEntry object (in file system)
zip.createWriter(new zip.FileWriter(fileEntry), function(writer) {

  zipWriter = writer;
  // ...

}, function(error) {
  // onerror callback
});
    • add a file into the zip 
      // use a BlobReader to read the FileEntry object to add
zipWriter.add(fEntry.name, new zip.BlobReader(fEntry), function() {

  // onend callback

}, function(currentIndex, totalIndex) {
  // onprogress callback
});
    • close the zip and get file content 
      zipWriter.close(function(blob) {

  // blob contains the zip file as a Blob object
  // or
  // var blobURL = fEntry.toURL(); 
  // blobURL contains the zip file as a URL

});

  2. Read a zip from a Blob object

    • get a ZipReader object to read a zip 
      var zipReader; // the ZipReader object
// use a BlobReader to read the zip from the FileEntry object
zip.createReader(new zip.BlobReader(fileEntry), function(reader) {

  zipReader = reader;
  // ...

}, function(error) {
  // onerror callback
});
    • get all entries from the zip 
      var zipEntry; // a Entry object
zipReader.getEntries(function(entries) {

  if (entries.length)
    zipEntry = entries[0];

});
    • get entry content 
      // use a BlobWriter to store the content into a Blob object (in RAM)
zipEntry.getData(new zip.BlobWriter(), function(blob) {

  // blob contains the zip file as a Blob object

}, function(current, total) {
  // onprogress callback
});
    • close the zip 
      zipReader.close(function() {

  // onclose callback

});
