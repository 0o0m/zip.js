zip.js

A JavasScript library to zip and unzip files

zip.js offers a low-level API for writing and reading large zip files (up to 4GB) with a stable RAM use.

This library is implemented with these HTML5 features:

  • Web workers
  • FileReader (and FileWriter optionally)
  • Blob and BlobBuilder
  • Typed arrays

Notes

  • This library is currently only supported with chrome 14 but should work with firefox when needed API will be implemented.

Install

  • include zip.js, deflate.js, and inflate.js file in your project
  • add zip.js in your HTML page 
    <script type="text/javascript" src="/lib/zip.js"></script>
  • set WORKER_SCRIPTS_PATH value in zip.js if needed 
    var WORKER_SCRIPTS_PATH = "/lib/";

API

  1. Write a zip into a File object

    • get a ZipWriter object to create the zip 
      // use a FileWriter to store the zip into a File object
var fileWriter = new zip.FileWriter(zipFile);
var zipWriter; // the ZipWriter object
zip.createWriter(fileWriter, false, function(writer) {

  zipWriter = writer;
  // ...

}, function(error) {
  // onerror callback
});
    • add a file into the zip 
      // use a BlobReader to read a File object
var blobReader = new zip.BlobReader(file);
zipWriter.add(file.name, blobReader, null, function() {

  // onend callback

}, function(currentIndex, totalIndex) {
  // onprogress callback
}, function(error) {
  // onerror callback
});
    • close the zip 
      fileWriter.close(function() {

  // onclose callback

}, function(error) {
  // onerror callback
});

  2. Read a zip from a Blob object

    • get a ZipReader object to read a zip 
      // use a BlobReader to read the zip from a File object
var blobReader = new zip.BlobReader(file);
var zipReader; // the ZipReader object
zip.createReader(blobReader, function(reader) {

  zipReader = reader;
  // ...

}, function(error) {
  // onerror callback
});
    • get all entries from the zip 
      var zipEntry; // a ZipEntry object
zipReader.getEntries(function(entries) {

  if (entries.length)
    zipEntry = entries[0];

}, function(error) {
  // onerror callback
});
    • get entry content 
      // use a FileWriter to store the content into a File object
var fileWriter = new zip.FileWriter(file);
zipEntry.getData(fileWriter, function() {

  // onend callback

}, function(current, total) {
  // onprogress callback
}, function(error) {
  // onerror callback
});
    • close the zip 
      zipReader.close(function() {

  // onclose callback

}, function(error) {
  // onerror callback
});
