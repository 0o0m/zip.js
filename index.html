zip.js

A new JavasScript library for zipping and unzipping files

zip.js offers a low-level API for writing and reading (big) zip files

This library is implemented with these HTML5 features:

  • Web workers
  • FileReaderSync and FileWriter
  • Blob and BlobBuilder
  • Typed arrays

Note

This library is currently only supported with chrome but will work with firefox when these features will be implemented.

Install

  • include zip.js, crc32.js, deflate.js, and inflate.js file in your project
  • add zip.js in your HTML page 
    <script type="text/javascript" src="/lib/zip.js"></script>
  • set WORKER_SCRIPTS_PATH value in zip.js if needed 
    var WORKER_SCRIPTS_PATH = "/lib/";

API

  • get a ZipWriter object to create a new zip file into a Blob object 
    var resourceWriter = new zip.BlobResourceWriter();
resourceWriter.init(blob, function() {
  // onsuccess callback
  var zipper = zip.createWriter(resourceWriter);
  // ...
}, function(error) {
  // onerror callback
});
  • add a file into the zip 
    zipper.add(filename, blob, null /* options */, function() {
  // onend callback
}, function(currentIndex, totalIndex) {
  // onprogress callback
}, function(error) {
  // error callback
});
  • close the zip 
    zipper.close(function() {
  // onclose callback
}, function(error) {
  // onerror callback
});
  • get the zip as a Blob 
    var blob = zipper.getBlob();

  • get a ZipReader object to read a zip file from a Blob object 
    var resourceReader = new zip.BlobResourceReader();
resourceReader.init(blob, function() {
  // onsuccess callback
  var unzipper = zip.createReader(resourceReader);
  // ...
}, function(error) {
  // onerror callback
});
  • get a ZipReader object to read a zip file from a given URL 
    var httpResourceReader = new zip.HttpRangeResourceReader();
httpResourceReader.init(URL, function() {
  // onsuccess callback
  var unzipper = zip.createReader(httpResourceReader);
  // ...
});
  • get all entries from the zip 
    unzipper.getEntries(function(entries) {
  // onsuccess callback
  //   entries parameter is an Array of Entry objects
}, function(error) {
  // onerror callback
});
  • get entry content as a Blob 
    entry.getData(function(blob) {
  // onend callback
  //   data parameter contains the file entry content
}, function(current, total) {
  // onprogress callback
}, function(error) {
  // onerror callback
});
  • close the zip 
    unzipper.close(function() {
  // onclose callback
}, function(error) {
  // onerror callback
});
